# SP6: CLI Integration — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Wire all `llm-rag` CLI commands to the agent layer built in SP1–SP5 via a single Typer app in `src/llm_rag/cli.py`.

**Architecture:** A single `src/llm_rag/cli.py` with a global `--debug` flag and 9 sub-commands. Async commands (`ask`, `ingest`, `run`, `fetch`, `compile-wiki`, `build-graph`) use `asyncio.run()`. Rich renders the `ask` answer as markdown. The `ingest`, `compile-wiki`, and `build-graph` commands use `PipelineRunner` as an async context manager (`async with PipelineRunner() as runner:`).

**Tech Stack:** `typer[all]>=0.12.0`, `rich` (bundled with typer[all]), `asyncio`, `pytest` with `typer.testing.CliRunner`, `unittest.mock`.

```python
from __future__ import annotations

import asyncio
import sys
from pathlib import Path
from typing import Any

import typer
from rich.console import Console
from rich.markdown import Markdown

from llm_rag.config import Settings, get_settings
from llm_rag.mcp.pool import MCPPool
from llm_rag.query.agent import QueryAgent

app = typer.Typer(help="Battery Research OS CLI")
console = Console()
_debug: bool = False


@app.callback()
def _main(debug: bool = typer.Option(False, "--debug", help="Show full tracebacks")) -> None:
    global _debug
    _debug = debug


# ---------------------------------------------------------------------------
# Shared helpers
# ---------------------------------------------------------------------------

def _load_all_manifests(raw_dir: Path) -> list[dict[str, Any]]:
    import json
    results = []
    for p in raw_dir.rglob("*.manifest.json"):
        try:
            results.append(json.loads(p.read_text()))
        except Exception:
            pass
    return results


def _find_source_by_doc_id(raw_dir: Path, doc_id: str) -> Path | None:
    for m in _load_all_manifests(raw_dir):
        if m.get("doc_id") == doc_id:
            return Path(m["source_path"])
    return None


def _load_topics(settings: Settings) -> list[str]:
    import yaml
    sources_yaml = settings.config_dir / "sources.yaml"
    if not sources_yaml.exists():
        return []
    data = yaml.safe_load(sources_yaml.read_text())
    return data.get("research_topics", [])


# ---------------------------------------------------------------------------
# ask
# ---------------------------------------------------------------------------

@app.command()
def ask(
    query: str = typer.Argument(..., help="Question to ask"),
    verbose: bool = typer.Option(False, "--verbose", help="Show sources"),
) -> None:
    """Ask a question against the knowledge base."""
    try:
        asyncio.run(_ask(query, verbose))
    except Exception as exc:
        if _debug:
            raise
        console.print(f"[red]Error:[/red] {exc}")
        raise typer.Exit(1)


async def _ask(query: str, verbose: bool) -> None:
    settings = get_settings()
    async with MCPPool() as pool:
        agent = QueryAgent(settings=settings)
        result = await agent.ask(query, pool)
    console.print(Markdown(result.answer))
    if verbose and result.sources:
        console.print("\n[bold]Sources[/bold]")
        for src in result.sources:
            console.print(f"  [dim]·[/dim] {src}")
```

# ---------------------------------------------------------------------------
# ingest
# ---------------------------------------------------------------------------

@app.command()
def ingest(
    path: Path | None = typer.Option(None, "--path", help="Directory or file to ingest"),
    doc_id: str | None = typer.Option(None, "--doc-id", help="Process one doc by ID"),
    force: bool = typer.Option(False, "--force", help="Reprocess regardless of content hash"),
) -> None:
    """Ingest changed files from raw/ into the pipeline."""
    try:
        settings = get_settings()

        if doc_id:
            source = _find_source_by_doc_id(settings.raw_dir, doc_id)
            if source is None:
                console.print(f"[red]Error:[/red] No file found for doc_id {doc_id!r}")
                raise typer.Exit(1)
            paths = [source]
        elif path:
            paths = [
                p for p in path.rglob("*")
                if p.is_file() and not p.name.endswith(".manifest.json")
            ]
        else:
            paths = [
                p for p in settings.raw_dir.rglob("*")
                if p.is_file() and not p.name.endswith(".manifest.json")
            ]

        asyncio.run(_ingest(paths, settings, force))
    except typer.Exit:
        raise
    except Exception as exc:
        if _debug:
            raise
        console.print(f"[red]Error:[/red] {exc}")
        raise typer.Exit(1)


async def _ingest(paths: list[Path], settings: Settings, force: bool) -> None:
    async with PipelineRunner(settings=settings) as runner:
        for p in paths:
            console.print(f"Ingesting [cyan]{p.name}[/cyan]")
            await runner.run(p, force=force)
```

@app.command()
def status() -> None:
    """Show corpus statistics and pipeline stage counts."""
    try:
        settings = get_settings()
        manifests = _load_all_manifests(settings.raw_dir)

        stage_counts: dict[str, int] = {}
        for m in manifests:

        table.add_column("Count", justify="right")

@app.command()
def run(
    interval: int = typer.Option(60, "--interval", help="Poll interval in seconds"),
) -> None:
    """Start the supervisor loop (Ctrl+C to stop)."""
    try:
        logging.basicConfig(
            level=logging.INFO,
            format="%(message)s",
            handlers=[RichHandler(console=console, rich_tracebacks=_debug, show_path=False)],
        )
        settings = get_settings()
        agent = SupervisorAgent(
            raw_dir=settings.raw_dir,
            settings=settings,
            interval_seconds=interval,
        )
        console.print("[green]Supervisor starting...[/green] Press Ctrl+C to stop.")
        asyncio.run(agent.run())
    except KeyboardInterrupt:
        console.print("\n[yellow]Supervisor stopped.[/yellow]")
    except Exception as exc:
        if _debug:
            raise
        console.print(f"[red]Error:[/red] {exc}")
        raise typer.Exit(1)

# ---------------------------------------------------------------------------
# fetch
# ---------------------------------------------------------------------------

_SEARCH_SUBAGENT_NAMES = ("arxiv", "semantic_scholar", "openalex", "pubmed")


@app.command()
def fetch(
    topic: str | None = typer.Option(None, "--topic", help="Search all sources for topic"),
    doi: str | None = typer.Option(None, "--doi", help="Fetch a specific paper by DOI"),
    url: str | None = typer.Option(None, "--url", help="Fetch a URL via Firecrawl"),
    source: str | None = typer.Option(None, "--source", help="Use only one source subagent"),
) -> None:
    """Fetch research papers from external sources."""
    try:
        asyncio.run(_fetch(topic=topic, doi=doi, url=url, source=source))
    except Exception as exc:
        if _debug:
            raise
        console.print(f"[red]Error:[/red] {exc}")
        raise typer.Exit(1)


async def _fetch(
    topic: str | None,
    doi: str | None,
    url: str | None,
    source: str | None,
) -> None:
    settings = get_settings()
    inbox = settings.raw_dir / "inbox"
    inbox.mkdir(parents=True, exist_ok=True)

    if doi:
        candidate = CandidateDocument(title=doi, abstract="", source="unpaywall", doi=doi)
        subagent = UnpaywallSubagent(email=settings.unpaywall_email)
        result = await subagent.fetch(candidate)
        if result is None:
            console.print(f"[yellow]No open-access PDF found for DOI {doi!r}[/yellow]")
        else:
            content, ext = result
            out = inbox / f"{doi.replace('/', '-')}.{ext}"
            out.write_bytes(content)
            console.print(f"Saved to [cyan]{out}[/cyan]")
        return

    if url:
        candidate = CandidateDocument(title=url, abstract="", source="firecrawl", source_url=url)
        subagent = FirecrawlSubagent(api_key=settings.firecrawl_api_key)
        result = await subagent.fetch(candidate)
        if result is None:
            console.print(f"[yellow]Could not fetch URL {url!r}[/yellow]")
        else:
            content, ext = result
            slug = url.split("//")[-1].replace("/", "-")[:60]
            out = inbox / f"{slug}.{ext}"
            out.write_bytes(content)
            console.print(f"Saved to [cyan]{out}[/cyan]")
        return

    subagent_map: dict[str, Any] = {
        "arxiv": ArXivSubagent(),
        "semantic_scholar": SemanticScholarSubagent(),
        "openalex": OpenAlexSubagent(),
        "pubmed": PubMedSubagent(),
    }

    if source:
        if source not in subagent_map:
            console.print(
                f"[red]Error:[/red] Unknown source {source!r}."
                f" Valid: {', '.join(subagent_map)}"
            )
            raise typer.Exit(1)
        subagents = [subagent_map[source]]
    else:
        subagents = list(subagent_map.values())

    topics = [topic] if topic else _load_topics(settings)
    agent = ResearchAgent(settings=settings, subagents=subagents)
    written = await agent.run(topics=topics)
    console.print(f"Fetched [bold]{len(written)}[/bold] document(s)")
```

## Task 6: `compile-wiki`, `build-graph`, `lint-wiki`, `export-graph` commands

**Files:**
- Modify: `src/llm_rag/cli.py`

Context:
- `compile-wiki` and `build-graph` use `PipelineRunner` (async context manager); they scan all manifests and re-run the pipeline for docs missing the target stage — `PipelineRunner.run()` is idempotent, it skips already-completed stages
- `lint-wiki` scans wiki markdown files for unclosed fence tags. `parse_page()` in `llm_rag.wiki.reader` does NOT raise on broken fences — it just silently misses them. Instead, use regex to check that every `<!-- auto-start: X -->` has a matching `<!-- auto-end: X -->` and vice versa
- `export-graph` loads the GraphML snapshot and copies it to `graph/exports/export.graphml`. Since the snapshot IS a GraphML file, `shutil.copy2` is the right tool
- `GraphStore` does not expose a public `.graph` property — access `store._g` is unnecessary; just copy the file

# ---------------------------------------------------------------------------
# lint-wiki

@app.command("lint-wiki")
def lint_wiki(
    fix: bool = typer.Option(False, "--fix", help="Auto-fix repairable issues (not yet implemented)"),
) -> None:
    """Check wiki pages for broken section fences."""
    try:
        settings = get_settings()
        issues = 0
        for md_file in sorted(settings.wiki_dir.rglob("*.md")):
            content = md_file.read_text()
            rel = md_file.relative_to(settings.root_dir)
            for fence_type in ("auto", "human"):
                starts = set(re.findall(rf"<!-- {fence_type}-start: (\S+) -->", content))
                ends = set(re.findall(rf"<!-- {fence_type}-end: (\S+) -->", content))
                for name in starts - ends:
                    issues += 1
                    console.print(f"[yellow]WARN[/yellow] {rel}: unclosed {fence_type}-start: {name!r}")
                    if fix:
                        console.print(f"  [dim]→ auto-fix not yet implemented[/dim]")
                for name in ends - starts:
                    issues += 1
                    console.print(f"[yellow]WARN[/yellow] {rel}: orphaned {fence_type}-end: {name!r}")
        console.print(f"\n[bold]{issues}[/bold] issue(s) found")
    except Exception as exc:
        if _debug:
            raise
        console.print(f"[red]Error:[/red] {exc}")
        raise typer.Exit(1)
```

@app.command("export-graph")
def export_graph(
    format: str = typer.Option("graphml", "--format", help="Export format: graphml or cypher"),
) -> None:
    """Export the knowledge graph snapshot."""
    try:
        settings = get_settings()
        snapshot = settings.graph_dir / "snapshots" / "latest.graphml"
        if not snapshot.exists():
            console.print(
                "[red]Error:[/red] No graph snapshot found."
                " Run `llm-rag build-graph` first."
            )
            raise typer.Exit(1)

        if format == "graphml":
            out = settings.graph_dir / "exports" / "export.graphml"
            out.parent.mkdir(parents=True, exist_ok=True)
            shutil.copy2(snapshot, out)
            console.print(f"Exported GraphML to [cyan]{out}[/cyan]")
        elif format == "cypher":
            console.print("[red]Error:[/red] Cypher export not yet implemented")
            raise typer.Exit(1)
        else:
            console.print(
                f"[red]Error:[/red] Unknown format {format!r}. Use graphml or cypher."
            )
            raise typer.Exit(1)
    except typer.Exit:
        raise
    except Exception as exc:
        if _debug:
            raise
        console.print(f"[red]Error:[/red] {exc}")
        raise typer.Exit(1)
```
